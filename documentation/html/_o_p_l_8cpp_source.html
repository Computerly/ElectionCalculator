79

88 setNumberOfSeats( max(getNumSeats() - nSeats, 0) );
89 i.setRemainder(i.getNumVotes() % getDroopQuota());
90 i.log( i.getNumSeats() );

92 }

97
98 sort(vParties.begin(), vParties.end(), [](Party &a, Party &b) {

100 return Election::flipCoin();
101 });
102

104 // * Distribute remaining seats *
105 // ****************************************************************************
106
107 for(Party &i : vParties){
108 i.incrementSeats( min(getNumSeats(), 1) );
